#version 450 core

layout(location = 0) in vec2 position;

out vec2 textureCoords;

uniform mat4 projectionMatrix;
uniform mat4 viewMatrix;
uniform mat4 modelMatrix;

uniform vec4 plane;
void main() {
    // Combine the matrices to get the final transformation

     vec2 scaledPosition = vec2(position.x *6.0, position.y * 6.0);

    // Combine the matrices to get the final transformation
    vec4 worldPosition = modelMatrix * vec4(scaledPosition.x, 0.0, scaledPosition.y, 1.0);

    gl_ClipDistance[0] = dot(worldPosition, plane);
    // Extrude in the Y-axis
   

    gl_Position = projectionMatrix * viewMatrix * worldPosition;
    textureCoords = vec2(position.x / 2.0 + 0.5, position.y / 2.0 + 0.5);

}
